設定縮放/符合高度
```python
from pypdf import PdfReader, PdfWriter
from pypdf.generic import NameObject, ArrayObject, NumberObject

# 載入原始 PDF
reader = PdfReader("input.pdf")
writer = PdfWriter()

# 將所有頁面加入新 PDF
for page in reader.pages:
    writer.add_page(page)

# 指定要設定的頁面（例如第 1 頁）
page_ref = writer.pages[0].indirect_reference

# 設定 OpenAction（開啟時視圖模式）
# 可選項：/Fit、/FitH、/FitV、/XYZ（自訂位置與縮放）
open_action = ArrayObject([
    page_ref,             # 頁面引用
    NameObject("/FitV"),  # 或 /Fit, /FitH
    NumberObject(0)       # 起始座標（/FitV 用 Y 座標）
])

# 更新 PDF 的 root 對象，加上 OpenAction
writer._root_object.update({
    NameObject("/OpenAction"): open_action
})

# 輸出新的 PDF 檔案
with open("output.pdf", "wb") as f:
    writer.write(f)

print("PDF 已加入 OpenAction 設定！")

```

整合右鍵轉換
```python
import sys
import os
import fitz  # PyMuPDF
from pypdf import PdfReader, PdfWriter
from pypdf.generic import NameObject, ArrayObject, NumberObject

def images_to_pdf(image_paths, output_pdf_path):
    pdf = fitz.open()
    for img_path in image_paths:
        try:
            img = fitz.Pixmap(img_path)
            page = pdf.new_page(width=img.width, height=img.height)
            rect = fitz.Rect(0, 0, img.width, img.height)
            page.insert_image(rect, filename=img_path)
        except Exception as e:
            print(f"⚠️ 跳過 {img_path}: {e}")
    pdf.save(output_pdf_path)
    pdf.close()

def set_pdf_open_action_fitv(input_pdf_path, output_pdf_path):
    reader = PdfReader(input_pdf_path)
    writer = PdfWriter()

    for page in reader.pages:
        writer.add_page(page)

    # 設定 OpenAction 為（FitV）要修改在 NameObject("/FitV")
    first_page_ref = writer.pages[0].indirect_reference
    open_action = ArrayObject([
        first_page_ref,
        NameObject("/FitV"),
        NumberObject(0)
    ])
    writer._root_object.update({
        NameObject("/OpenAction"): open_action
    })

    with open(output_pdf_path, "wb") as f:
        writer.write(f)

if __name__ == "__main__":
    image_paths = sys.argv[1:]
    if not image_paths:
        print("請提供圖片檔案")
        sys.exit(1)

    # 排序檔案（按檔名）
    image_paths.sort()

    # 暫存 PDF 檔案（由 PyMuPDF 生成）
    output_folder = os.path.dirname(image_paths[0])
    temp_pdf = os.path.join(output_folder, "_temp_output.pdf")
    final_pdf = os.path.join(output_folder, "merged.pdf")

    images_to_pdf(image_paths, temp_pdf)
    set_pdf_open_action_fitv(temp_pdf, final_pdf)

    os.remove(temp_pdf)
    os.startfile(final_pdf)
```

修改版本 png使用pillow jpg使用img2pdf，最後用pypdf改openation

```python
# 執行前安裝依賴 pip install img2pdf reportlab pillow pypdf==3.16.4
import os
import sys
import img2pdf
from PIL import Image
from io import BytesIO
from reportlab.pdfgen import canvas
from reportlab.lib.pagesizes import portrait
from reportlab.lib.utils import ImageReader
from pypdf import PdfReader, PdfWriter
from pypdf.generic import NameObject, ArrayObject, NumberObject

def convert_jpgs_to_pdf(image_paths):
    jpgs = [img for img in image_paths if img.lower().endswith((".jpg", ".jpeg"))]
    return BytesIO(img2pdf.convert(jpgs)) if jpgs else None

def convert_pngs_to_pdf(image_paths):
    pngs = [img for img in image_paths if img.lower().endswith(".png")]
    if not pngs:
        return None
    output = BytesIO()
    c = canvas.Canvas(output)
    for img_path in pngs:
        with Image.open(img_path) as img:
            width, height = img.size
            c.setPageSize(portrait((width, height)))
            c.drawImage(ImageReader(img), 0, 0, width=width, height=height)
            c.showPage()
    c.save()
    output.seek(0)
    return output

def merge_pdfs(jpg_pdf_stream, png_pdf_stream):
    merger = PdfWriter()
    if jpg_pdf_stream:
        reader = PdfReader(jpg_pdf_stream)
        for page in reader.pages:
            merger.add_page(page)
    if png_pdf_stream:
        reader = PdfReader(png_pdf_stream)
        for page in reader.pages:
            merger.add_page(page)
    output = BytesIO()
    merger.write(output)
    output.seek(0)
    return output

def set_pdf_open_action_fitv(pdf_stream, output_path):
    reader = PdfReader(pdf_stream)
    writer = PdfWriter()
    writer.clone_document_from_reader(reader)

    # 設定開啟時符合頁面高度（FitV）
    first_page_ref = writer.pages[0].indirect_reference
    open_action = ArrayObject([
        first_page_ref,
        NameObject("/FitV"),
        NumberObject(0)
    ])
    writer._root_object.update({
        NameObject("/OpenAction"): open_action
    })

    with open(output_path, "wb") as f:
        writer.write(f)

def save_pdf(pdf_stream, output_path):
    with open(output_path, "wb") as f:
        f.write(pdf_stream.read())

if __name__ == "__main__":
    image_paths = sys.argv[1:]
    if not image_paths:
        print("請拖曳圖片檔案到此執行檔")
        sys.exit(1)

    image_paths.sort()
    folder = os.path.dirname(image_paths[0])
    output_pdf = os.path.join(folder, "merged.pdf")

    # 將 JPG 和 PNG 轉換為 PDF
    jpg_pdf = convert_jpgs_to_pdf(image_paths)
    png_pdf = convert_pngs_to_pdf(image_paths)

    # 合併 JPG 和 PNG PDF
    merged_stream = merge_pdfs(jpg_pdf, png_pdf)

    # 設定 /OpenAction 為 FitV，並保存最終 PDF
    set_pdf_open_action_fitv(merged_stream, output_pdf)

    print(f"✅ 已產生 PDF 並設定 FitV 開啟動作：{output_pdf}") 
    os.startfile(output_pdf) 
```

Pillow 無法保留 PNG 壓縮器（如 pngquant）所做的極致壓縮，因為- Pillow 讀取圖片時會「解碼」成完整圖像資料（RGBA）完全不重編碼，直接二進位合併
