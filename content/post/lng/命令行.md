自動備份檔案到NAS 腳本
https://youtu.be/E0PDhX2r5yg?si=3IIHaCYZfKFSaPyV

```shell:backup.bat
@echo off
set _DAY=%DATE:~-1%
set _ZIP=mysql-%_DAY%.7z
set _DUMP=d:\dump\mysqldump-%_DAY%.sql
echo %DATE% %TIME% START dump file=%_DUMP%, zip=%_ZIP% >> d:\auto\backup_mysql.log
mysqldump -u root -p your_password --all-databases > %_DUMP%
echo %TIME% After dump >> d:\auto\backup_mysql.log
"c:\Program Files\7-zip\7z.exe" a d:\backup\%_ZIP% %_DUMP%
echo %TIME% After 7z >> d:\auto\backup_mysql.log
xcopy d:\backup\%_ZIP% s:\backup\mysql\%_ZIP% /y
echo %TIME% After xcopy >> d:\auto\backup_mysql.log
if exist s:\backup\mysql\%_ZIP% del d:\backup\%_ZIP%
echo %DATE %TIME END >> d:\auto\backup_mysql.log
```
